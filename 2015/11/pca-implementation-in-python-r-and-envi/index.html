e = IDL.ENVI()

% Restored file: ENVI. % Loaded DLM: HPGRAPHICS. % Loaded DLM: XML. % Loaded DLM: PNG.

new_raster = IDL.ENVIRaster(cropB12_spc_reshape)
new_raster.save()

% Loaded DLM: NATIVE.

Task = IDL.ENVITask('ForwardPCATransform') 
Task.INPUT_RASTER = new_raster 
Task.OUTPUT_RASTER_URI = e.GetTemporaryFilename() 
Task.Execute()

% Loaded DLM: LAPACK.

PCA in R

import numpy as np
from rpy2 import robjects
from rpy2.robjects import pandas2ri
pandas2ri.activate()

# http://gastonsanchez.com/blog/how-to/2012/06/17/PCA-in-R.html
# the PCA is supposed to be performed with standardized data (mean = 0, variance = 1).
#However, in the following methods, the scale in set to false.

robjects.r('''
        pca1 <- function(data){
        pca1 = prcomp(data, scale. = FALSE)
        return(pca1$x)
        }

        pca2 <- function(data){
        pca2 = princomp(data, cor = FALSE)
        return(pca2$scores)
        }

        pca3 <- function(data){
        library(FactoMineR)
            pca3 = PCA(data, graph = FALSE,scale.unit = FALSE)
        return(pca3)
        }

    ''')
